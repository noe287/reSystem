SKY-FW Install 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------  

- Stop the device during bootup with ctrl-c 

- Assuming that the firmware cited below are in the "tftpboot" folder on your computer. 

  cat /airties_user/etc/data.xml

BOOSTER 

f 192.168.2.77:bcmEXTENDER_fs_kernel 

f 192.168.2.77:bcmEXTENDER_fs_kernel_local_build 

-----------------------------------------------------------------------------------------------------------------------------------------------------------------  

VIPER 

f 192.168.2.77:bcmBSKYB_VIPER_fs_kernel 
f 192.168.2.77:bcmBSKYB_VIPER_fs_kernel_non 

f 192.168.2.77:bcmBSKYB_VIPER_fs_kernel_18AP 

bcmBSKYB_VIPER_fs_kernel_smart 

-----------------------------------------------------------------------------------------------------------------------------------------------------------------

MRBOX  

tftp 40000000 192.168.2.77:uImage;mmc write 40000000 0 14000; reset 
tftp 40000000 192.168.2.77:uImage_non;mmc write 40000000 0 14000; reset 


tftp 40000000 192.168.2.77:uImageN_eng_release;mmc write 40000000 0 14000; reset 

tftp 40000000 192.168.2.77:uImageNmr;mmc write 40000000 0 14000; reset 

tftp 40000000 192.168.2.77:uImageN_eng_release;mmc write 40000000 0 14000; reset 

tftp 40000000 192.168.2.77:1077mr;mmc write 40000000 0 14000; reset 

-----------------------------------------------------------------------------------------------------------------------------------------------------------------  

XWING 

tftp 40000000 192.168.2.77:uImage; mmc write 40000000 0 14000; reset 

tftp 40000000 192.168.2.77:uImageNXwing; mmc write 40000000 0 14000; reset 

tftp 40000000 192.168.2.77:uImageNX_eng_release; mmc write 40000000 0 14000; reset 

tftp 40000000 192.168.2.77:uImageX; mmc write 40000000 0 14000; reset 

-----------------------------------------------------------------------------------------------------------------------------------------------------------------

FALCON  

ifconfig eth0 -addr=192.168.2.7 -mask=255.255.255.0 -gw=192.168.2.77 -dns=192.168.2.77;flash 192.168.2.77:zImage emmcflash0.kernel;reboot; 

ifconfig eth0 -addr=192.168.2.7 -mask=255.255.255.0 -gw=192.168.2.77 -dns=192.168.2.77;flash 192.168.2.77:zImageOrg emmcflash0.kernel;reboot; 

-----------------------------------------------------------------------------------------------------------------------------------------------------------------  

GEMINI 

setenv ipaddr 192.168.2.50; tftp 42008000 192.168.2.77:zImage; mmc write 42008000 52000 19000; reset
